Validação Simples de Formulário

$("#form1").validate({
	redirect: false,
	onValidate: function(){
		alert( $(this).serialize() );
	}
});







Homem Mulher


Formulário via Ajax

$('button').click(function(){

	$.ajax({
		url: 'ajax/validate.html',
		success: function(data){
			$('#live').prepend(data);

			// The trick
			$('#form2').validate();
		}
	});
});

$('#form3').validate({
