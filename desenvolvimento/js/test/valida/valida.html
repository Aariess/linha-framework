Validação Simples de Formulário

Homem Mulher

$(".formulario").valida({
redireciona: false,
live: true,
onValida: function(){
$(this).removeClass('erro').parent().removeClass('erro');
},
onErro: function(){
if($(this).is(':checkbox') || $(this).is(':radio')){
$(this).parent().addClass('erro');
}else{
$(this).addClass('erro');
}
},
onPassa: function(){
alert($(this).serialize());
}
});
