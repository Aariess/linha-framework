Validação Simples de Formulário

	$("#formulario1").valida({
		redireciona: false,
		onValida: function(){
			$(this).parent().removeClass('erro');
		},
		onErro: function(){
			if($(this).is(':checkbox') || $(this).is(':radio')){
				$(this).parent().addClass('erro');
			}
		},
		onPassa: function(){
			alert($(this).serialize());
		}
	});
